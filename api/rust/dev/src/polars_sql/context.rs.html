125    /// Execute a SQL query, returning a [`LazyFrame`].
126    /// ```rust
127    /// # use polars_sql::SQLContext;
128    /// # use polars_core::prelude::*;
129    /// # use polars_lazy::prelude::*;
130    /// # fn main() {
131    ///
132    /// let mut ctx = SQLContext::new();
133    /// let df = df! {
134    ///    "a" =>  [1, 2, 3],
135    /// }
136    /// .unwrap();
137    ///
138    /// ctx.register("df", df.clone().lazy());
139    /// let sql_df = ctx.execute("SELECT * FROM df").unwrap().collect().unwrap();
140    /// assert!(sql_df.equals(&df));
141    /// # }
142    ///```
143    pub fn execute(&mut self, query: &str) -> PolarsResult<LazyFrame> {
144        let mut parser = Parser::new(&GenericDialect);
145        parser = parser.with_options(ParserOptions {
146            trailing_commas: true,
147            ..Default::default()
148        });
149
150        let ast = parser
151            .try_with_sql(query)
152            .map_err(to_sql_interface_err)?
153            .parse_statements()
154            .map_err(to_sql_interface_err)?;
155
156        polars_ensure!(ast.len() == 1, SQLInterface: "one (and only one) statement can be parsed at a time");
157        let res = self.execute_statement(ast.first().unwrap())?;
158
159        // Ensure the result uses the proper arenas.
160        // This will instantiate new arenas with a new version.
161        let lp_arena = std::mem::take(&mut self.lp_arena);
162        let expr_arena = std::mem::take(&mut self.expr_arena);
163        res.set_cached_arena(lp_arena, expr_arena);
164
165        // Every execution should clear the statement-level maps.
166        self.cte_map.clear();
167        self.table_aliases.clear();
168        self.joined_aliases.clear();
169
170        Ok(res)
171    }

192    pub(crate) fn execute_statement(&mut self, stmt: &Statement) -> PolarsResult<LazyFrame> {
193        let ast = stmt;
194        Ok(match ast {
195            Statement::Query(query) => self.execute_query(query)?,
196            stmt @ Statement::ShowTables { .. } => self.execute_show_tables(stmt)?,
197            stmt @ Statement::CreateTable { .. } => self.execute_create_table(stmt)?,

201            } => self.execute_drop_table(stmt)?,
202            stmt @ Statement::Explain { .. } => self.execute_explain(stmt)?,
203            stmt @ Statement::Truncate { .. } => self.execute_truncate_table(stmt)?,
204            stmt @ Statement::Delete { .. } => self.execute_delete_from_table(stmt)?,
205            _ => polars_bail!(
206                SQLInterface: "statement type is not supported:\n{:?}", ast,
207            ),
208        })

225    pub(super) fn get_table_from_current_scope(&self, name: &str) -> Option<LazyFrame> {
226        // Resolve the table name in the current scope; multi-stage fallback
227        // * table name → cte name
228        // * table alias → cte alias
229        self.table_map
230            .get(name)
231            .or_else(|| self.cte_map.get(name))
232            .or_else(|| {
233                self.table_aliases.get(name).and_then(|alias| {
234                    self.table_map
235                        .get(alias.as_str())
236                        .or_else(|| self.cte_map.get(alias.as_str()))
237                })
238            })
239            .cloned()
240    }

242    /// Execute a query in an isolated context. This prevents subqueries from mutating
243    /// arenas and other context state. Returns both the LazyFrame *and* its associated
244    /// Schema (so that the correct arenas are used when determining schema).
245    pub(crate) fn execute_isolated<F>(&mut self, query: F) -> PolarsResult<(LazyFrame, SchemaRef)>
246    where
247        F: FnOnce(&mut Self) -> PolarsResult<LazyFrame>,
248    {
249        // Save key state (arenas and lookups)
250        let (joined_aliases, table_aliases, lp_arena, expr_arena, table_map) = (
251            // "take" to ensure subqueries start with clean state
252            std::mem::take(&mut self.joined_aliases),
253            std::mem::take(&mut self.table_aliases),
254            std::mem::take(&mut self.lp_arena),
255            std::mem::take(&mut self.expr_arena),
256            // "clone" to allow subqueries to see registered tables
257            self.table_map.clone(),
258        );
259
260        // Execute query with clean state (eg: nested/subquery)
261        let mut lf = query(self)?;
262        let schema = self.get_frame_schema(&mut lf)?;
263
264        // Restore saved state
265        lf.set_cached_arena(
266            std::mem::replace(&mut self.lp_arena, lp_arena),
267            std::mem::replace(&mut self.expr_arena, expr_arena),
268        );
269        self.joined_aliases = joined_aliases;
270        self.table_aliases = table_aliases;
271        self.table_map = table_map;
272
273        Ok((lf, schema))
274    }

276    fn expr_or_ordinal(
277        &mut self,
278        e: &SQLExpr,
279        exprs: &[Expr],
280        selected: Option<&[Expr]>,
281        schema: Option<&Schema>,
282        clause: &str,
283    ) -> PolarsResult<Expr> {
284        match e {
285            SQLExpr::UnaryOp {
286                op: UnaryOperator::Minus,
287                expr,
288            } if matches!(**expr, SQLExpr::Value(SQLValue::Number(_, _))) => {
289                if let SQLExpr::Value(SQLValue::Number(ref idx, _)) = **expr {
290                    Err(polars_err!(
291                    SQLSyntax:
292                    "negative ordinal values are invalid for {}; found -{}",
293                    clause,
294                    idx
295                    ))
296                } else {
297                    unreachable!()
298                }
299            },
300            SQLExpr::Value(SQLValue::Number(idx, _)) => {
301                // note: sql queries are 1-indexed
302                let idx = idx.parse::<usize>().map_err(|_| {
303                    polars_err!(
304                        SQLSyntax:
305                        "negative ordinal values are invalid for {}; found {}",
306                        clause,
307                        idx
308                    )
309                })?;
310                // note: "selected" cols represent final projection order, so we use those for
311                // ordinal resolution. "exprs" may include cols that are subsequently dropped.
312                let cols = if let Some(cols) = selected {
313                    cols
314                } else {
315                    exprs
316                };
317                Ok(cols
318                    .get(idx - 1)
319                    .ok_or_else(|| {
320                        polars_err!(
321                            SQLInterface:
322                            "{} ordinal value must refer to a valid column; found {}",
323                            clause,
324                            idx
325                        )
326                    })?
327                    .clone())
328            },
329            SQLExpr::Value(v) => Err(polars_err!(
330                SQLSyntax:
331                "{} requires a valid expression or positive ordinal; found {}", clause, v,
332            )),
333            _ => parse_sql_expr(e, self, schema),
334        }
335    }

346    fn process_query(&mut self, expr: &SetExpr, query: &Query) -> PolarsResult<LazyFrame> {
347        match expr {
348            SetExpr::Select(select_stmt) => self.execute_select(select_stmt, query),
349            SetExpr::Query(query) => self.execute_query_no_ctes(query),
350            SetExpr::SetOperation {
351                op: SetOperator::Union,
352                set_quantifier,
353                left,
354                right,
355            } => self.process_union(left, right, set_quantifier, query),
356
357            #[cfg(feature = "semi_anti_join")]
358            SetExpr::SetOperation {
359                op: SetOperator::Intersect | SetOperator::Except,
360                set_quantifier,
361                left,
362                right,
363            } => self.process_except_intersect(left, right, set_quantifier, query),
364
365            SetExpr::Values(Values {
366                explicit_row: _,
367                rows,
368            }) => self.process_values(rows),
369
370            SetExpr::Table(tbl) => {
371                if tbl.table_name.is_some() {
372                    let table_name = tbl.table_name.as_ref().unwrap();
373                    self.get_table_from_current_scope(table_name)
374                        .ok_or_else(|| {
375                            polars_err!(
376                                SQLInterface: "no table or alias named '{}' found",
377                                tbl
378                            )
379                        })
380                } else {
381                    polars_bail!(SQLInterface: "'TABLE' requires valid table name")
382                }
383            },
384            op => {
385                polars_bail!(SQLInterface: "'{}' operation is currently unsupported", op)
386            },
387        }
388    }

390    #[cfg(feature = "semi_anti_join")]
391    fn process_except_intersect(
392        &mut self,
393        left: &SetExpr,
394        right: &SetExpr,
395        quantifier: &SetQuantifier,
396        query: &Query,
397    ) -> PolarsResult<LazyFrame> {
398        let (join_type, op_name) = match *query.body {
399            SetExpr::SetOperation {
400                op: SetOperator::Except,
401                ..
402            } => (JoinType::Anti, "EXCEPT"),
403            _ => (JoinType::Semi, "INTERSECT"),
404        };
405        let mut lf = self.process_query(left, query)?;
406        let mut rf = self.process_query(right, query)?;
407        let lf_schema = self.get_frame_schema(&mut lf)?;
408
409        let lf_cols: Vec<_> = lf_schema.iter_names_cloned().map(col).collect();
410        let rf_cols = match quantifier {
411            SetQuantifier::ByName => None,
412            SetQuantifier::Distinct | SetQuantifier::None => {
413                let rf_schema = self.get_frame_schema(&mut rf)?;
414                let rf_cols: Vec<_> = rf_schema.iter_names_cloned().map(col).collect();
415                if lf_cols.len() != rf_cols.len() {
416                    polars_bail!(SQLInterface: "{} requires equal number of columns in each table (use '{} BY NAME' to combine mismatched tables)", op_name, op_name)
417                }
418                Some(rf_cols)
419            },
420            _ => {
421                polars_bail!(SQLInterface: "'{} {}' is not supported", op_name, quantifier.to_string())
422            },
423        };
424        let join = lf.join_builder().with(rf).how(join_type).join_nulls(true);
425        let joined_tbl = match rf_cols {
426            Some(rf_cols) => join.left_on(lf_cols).right_on(rf_cols).finish(),
427            None => join.on(lf_cols).finish(),
428        };
429        Ok(joined_tbl.unique(None, UniqueKeepStrategy::Any))
430    }

432    fn process_union(
433        &mut self,
434        left: &SetExpr,
435        right: &SetExpr,
436        quantifier: &SetQuantifier,
437        query: &Query,
438    ) -> PolarsResult<LazyFrame> {
439        let mut lf = self.process_query(left, query)?;
440        let mut rf = self.process_query(right, query)?;
441        let opts = UnionArgs {
442            parallel: true,
443            to_supertypes: true,
444            ..Default::default()
445        };
446        match quantifier {
447            // UNION [ALL | DISTINCT]
448            SetQuantifier::All | SetQuantifier::Distinct | SetQuantifier::None => {
449                let lf_schema = self.get_frame_schema(&mut lf)?;
450                let rf_schema = self.get_frame_schema(&mut rf)?;
451                if lf_schema.len() != rf_schema.len() {
452                    polars_bail!(SQLInterface: "UNION requires equal number of columns in each table (use 'UNION BY NAME' to combine mismatched tables)")
453                }
454                // rename `rf` columns to match `lf` if they differ; SQL behaves
455                // positionally on UNION ops (unless using the "BY NAME" qualifier)
456                if lf_schema.iter_names().ne(rf_schema.iter_names()) {
457                    rf = rf.rename(rf_schema.iter_names(), lf_schema.iter_names(), true);
458                }
459                let concatenated = concat(vec![lf, rf], opts);
460                match quantifier {
461                    SetQuantifier::Distinct | SetQuantifier::None => {
462                        concatenated.map(|lf| lf.unique(None, UniqueKeepStrategy::Any))
463                    },
464                    _ => concatenated,
465                }
466            },
467            // UNION ALL BY NAME
468            #[cfg(feature = "diagonal_concat")]
469            SetQuantifier::AllByName => concat_lf_diagonal(vec![lf, rf], opts),
470            // UNION [DISTINCT] BY NAME
471            #[cfg(feature = "diagonal_concat")]
472            SetQuantifier::ByName | SetQuantifier::DistinctByName => {
473                let concatenated = concat_lf_diagonal(vec![lf, rf], opts);
474                concatenated.map(|lf| lf.unique(None, UniqueKeepStrategy::Any))
475            },
476            #[allow(unreachable_patterns)]
477            _ => polars_bail!(SQLInterface: "'UNION {}' is not currently supported", quantifier),
478        }
479    }

481    fn process_values(&mut self, values: &[Vec<SQLExpr>]) -> PolarsResult<LazyFrame> {
482        let frame_rows: Vec<Row> = values.iter().map(|row| {
483            let row_data: Result<Vec<_>, _> = row.iter().map(|expr| {
484                let expr = parse_sql_expr(expr, self, None)?;
485                match expr {
486                    Expr::Literal(value) => {
487                        value.to_any_value()
488                            .ok_or_else(|| polars_err!(SQLInterface: "invalid literal value: {:?}", value))
489                            .map(|av| av.into_static())
490                    },
491                    _ => polars_bail!(SQLInterface: "VALUES clause expects literals; found {}", expr),
492                }
493            }).collect();
494            row_data.map(Row::new)
495        }).collect::<Result<_, _>>()?;
496
497        Ok(DataFrame::from_rows(frame_rows.as_ref())?.lazy())
498    }

538    // DELETE FROM <tbl> [WHERE ...]
539    fn execute_delete_from_table(&mut self, stmt: &Statement) -> PolarsResult<LazyFrame> {
540        if let Statement::Delete(Delete {
541            tables,
542            from,
543            using,
544            selection,
545            returning,
546            order_by,
547            limit,
548        }) = stmt
549        {
550            if !tables.is_empty()
551                || using.is_some()
552                || returning.is_some()
553                || limit.is_some()
554                || !order_by.is_empty()
555            {
556                let error_message = match () {
557                    _ if !tables.is_empty() => "DELETE expects exactly one table name",
558                    _ if using.is_some() => "DELETE does not support the USING clause",
559                    _ if returning.is_some() => "DELETE does not support the RETURNING clause",
560                    _ if limit.is_some() => "DELETE does not support the LIMIT clause",
561                    _ if !order_by.is_empty() => "DELETE does not support the ORDER BY clause",
562                    _ => unreachable!(),
563                };
564                polars_bail!(SQLInterface: error_message);
565            }
566            let from_tables = match &from {
567                FromTable::WithFromKeyword(from) => from,
568                FromTable::WithoutKeyword(from) => from,
569            };
570            if from_tables.len() > 1 {
571                polars_bail!(SQLInterface: "cannot have multiple tables in DELETE FROM (found {})", from_tables.len())
572            }
573            let tbl_expr = from_tables.first().unwrap();
574            if !tbl_expr.joins.is_empty() {
575                polars_bail!(SQLInterface: "DELETE does not support table JOINs")
576            }
577            let (_, mut lf) = self.get_table(&tbl_expr.relation)?;
578            if selection.is_none() {
579                // no WHERE clause; equivalent to TRUNCATE (drop all rows)
580                Ok(DataFrame::empty_with_schema(
581                    lf.schema_with_arenas(&mut self.lp_arena, &mut self.expr_arena)
582                        .unwrap()
583                        .as_ref(),
584                )
585                .lazy())
586            } else {
587                // apply constraint as inverted filter (drops rows matching the selection)
588                Ok(self.process_where(lf.clone(), selection, true, None)?)
589            }
590        } else {
591            polars_bail!(SQLInterface: "unexpected statement type; expected DELETE")
592        }
593    }

595    // TRUNCATE <tbl>
596    fn execute_truncate_table(&mut self, stmt: &Statement) -> PolarsResult<LazyFrame> {
597        if let Statement::Truncate {
598            table_names,
599            partitions,
600            ..
601        } = stmt
602        {
603            match partitions {
604                None => {
605                    if table_names.len() != 1 {
606                        polars_bail!(SQLInterface: "TRUNCATE expects exactly one table name; found {}", table_names.len())
607                    }
608                    let tbl = table_names[0].to_string();
609                    if let Some(lf) = self.table_map.get_mut(&tbl) {
610                        *lf = DataFrame::empty_with_schema(
611                            lf.schema_with_arenas(&mut self.lp_arena, &mut self.expr_arena)
612                                .unwrap()
613                                .as_ref(),
614                        )
615                        .lazy();
616                        Ok(lf.clone())
617                    } else {
618                        polars_bail!(SQLInterface: "table '{}' does not exist", tbl);
619                    }
620                },
621                _ => {
622                    polars_bail!(SQLInterface: "TRUNCATE does not support use of 'partitions'")
623                },
624            }
625        } else {
626            polars_bail!(SQLInterface: "unexpected statement type; expected TRUNCATE")
627        }

635        if let Some(with) = &query.with {
636            if with.recursive {
637                polars_bail!(SQLInterface: "recursive CTEs are not supported")
638            }

649    /// execute the 'FROM' part of the query
650    fn execute_from_statement(&mut self, tbl_expr: &TableWithJoins) -> PolarsResult<LazyFrame> {
651        let (l_name, mut lf) = self.get_table(&tbl_expr.relation)?;
652        if !tbl_expr.joins.is_empty() {
653            for join in &tbl_expr.joins {
654                let (r_name, mut rf) = self.get_table(&join.relation)?;
655                if r_name.is_empty() {
656                    // Require non-empty to avoid duplicate column errors from nested self-joins.
657                    polars_bail!(
658                        SQLInterface:
659                        "cannot join on unnamed relation; please provide an alias"
660                    )
661                }
662                let left_schema = self.get_frame_schema(&mut lf)?;
663                let right_schema = self.get_frame_schema(&mut rf)?;
664
665                lf = match &join.join_operator {
666                    op @ (JoinOperator::FullOuter(constraint)
667                    | JoinOperator::LeftOuter(constraint)
668                    | JoinOperator::RightOuter(constraint)
669                    | JoinOperator::Inner(constraint)
670                    | JoinOperator::Anti(constraint)
671                    | JoinOperator::Semi(constraint)
672                    | JoinOperator::LeftAnti(constraint)
673                    | JoinOperator::LeftSemi(constraint)
674                    | JoinOperator::RightAnti(constraint)
675                    | JoinOperator::RightSemi(constraint)) => {
676                        let (lf, rf) = match op {
677                            JoinOperator::RightAnti(_) | JoinOperator::RightSemi(_) => (rf, lf),
678                            _ => (lf, rf),
679                        };
680                        self.process_join(
681                            &TableInfo {
682                                frame: lf,
683                                name: (&l_name).into(),
684                                schema: left_schema.clone(),
685                            },
686                            &TableInfo {
687                                frame: rf,
688                                name: (&r_name).into(),
689                                schema: right_schema.clone(),
690                            },
691                            constraint,
692                            match op {
693                                JoinOperator::FullOuter(_) => JoinType::Full,
694                                JoinOperator::LeftOuter(_) => JoinType::Left,
695                                JoinOperator::RightOuter(_) => JoinType::Right,
696                                JoinOperator::Inner(_) => JoinType::Inner,
697                                #[cfg(feature = "semi_anti_join")]
698                                JoinOperator::Anti(_)
699                                | JoinOperator::LeftAnti(_)
700                                | JoinOperator::RightAnti(_) => JoinType::Anti,
701                                #[cfg(feature = "semi_anti_join")]
702                                JoinOperator::Semi(_)
703                                | JoinOperator::LeftSemi(_)
704                                | JoinOperator::RightSemi(_) => JoinType::Semi,
705                                join_type => polars_bail!(
706                                    SQLInterface:
707                                    "join type '{:?}' not currently supported",
708                                    join_type
709                                ),
710                            },
711                        )?
712                    },
713                    JoinOperator::CrossJoin => {
714                        lf.cross_join(rf, Some(format_pl_smallstr!(":{}", r_name)))
715                    },
716                    join_type => {
717                        polars_bail!(SQLInterface: "join type '{:?}' not currently supported", join_type)
718                    },
719                };
720
721                // track join-aliased columns so we can resolve them later
722                let joined_schema = self.get_frame_schema(&mut lf)?;
723
724                self.joined_aliases.insert(
725                    r_name.clone(),
726                    right_schema
727                        .iter_names()
728                        .filter_map(|name| {
729                            // col exists in both tables and is aliased in the joined result
730                            let aliased_name = format!("{name}:{r_name}");
731                            if left_schema.contains(name)
732                                && joined_schema.contains(aliased_name.as_str())
733                            {
734                                Some((name.to_string(), aliased_name))
735                            } else {
736                                None
737                            }
738                        })
739                        .collect::<PlHashMap<String, String>>(),
740                );
741            }
742        };
743        Ok(lf)
744    }

746    /// Check that the SELECT statement only contains supported clauses.
747    fn validate_select(&self, select_stmt: &Select) -> PolarsResult<()> {
748        // Destructure "Select" exhaustively; that way if/when new fields are added in
749        // future sqlparser versions, we'll get a compilation error and can handle them
750        let Select {
751            // Supported clauses
752            distinct: _,
753            from: _,
754            group_by: _,
755            having: _,
756            projection: _,
757            selection: _,
758
759            // Metadata/token fields (can ignore)
760            select_token: _,
761            top_before_distinct: _,
762            window_before_qualify: _,
763
764            // Unsupported clauses
765            ref cluster_by,
766            ref connect_by,
767            ref distribute_by,
768            ref into,
769            ref lateral_views,
770            ref named_window,
771            ref prewhere,
772            ref qualify,
773            ref sort_by,
774            ref top,
775            ref value_table_mode,
776        } = *select_stmt;
777
778        // Raise specific error messages for unsupported attributes
779        polars_ensure!(cluster_by.is_empty(), SQLInterface: "CLUSTER BY clause is not supported");
780        polars_ensure!(connect_by.is_none(), SQLInterface: "CONNECT BY clause is not supported");
781        polars_ensure!(distribute_by.is_empty(), SQLInterface: "DISTRIBUTE BY clause is not supported");
782        polars_ensure!(into.is_none(), SQLInterface: "SELECT INTO clause is not supported");
783        polars_ensure!(lateral_views.is_empty(), SQLInterface: "LATERAL VIEW clause is not supported");
784        polars_ensure!(named_window.is_empty(), SQLInterface: "WINDOW clause support is coming soon");
785        polars_ensure!(prewhere.is_none(), SQLInterface: "PREWHERE clause is not supported");
786        polars_ensure!(qualify.is_none(), SQLInterface: "QUALIFY clause is not currently supported");
787        polars_ensure!(sort_by.is_empty(), SQLInterface: "SORT BY clause is not supported; use ORDER BY instead");
788        polars_ensure!(top.is_none(), SQLInterface: "TOP clause is not supported; use LIMIT instead");
789        polars_ensure!(value_table_mode.is_none(), SQLInterface: "SELECT AS VALUE/STRUCT is not supported");
790        Ok(())
791    }

793    /// Execute the 'SELECT' part of the query.
794    fn execute_select(&mut self, select_stmt: &Select, query: &Query) -> PolarsResult<LazyFrame> {
795        // Check that the statement doesn't contain unsupported SELECT clauses
796        self.validate_select(select_stmt)?;
797
798        let mut lf = if select_stmt.from.is_empty() {
799            DataFrame::empty().lazy()
800        } else {
801            // Note: implicit joins need more work to support properly,
802            // explicit joins are preferred for now (ref: #16662)
803            let from = select_stmt.clone().from;
804            if from.len() > 1 {
805                polars_bail!(SQLInterface: "multiple tables in FROM clause are not currently supported (found {}); use explicit JOIN syntax instead", from.len())
806            }
807            self.execute_from_statement(from.first().unwrap())?
808        };
809
810        // Filter expression (WHERE clause)
811        let schema = self.get_frame_schema(&mut lf)?;
812        lf = self.process_where(lf, &select_stmt.selection, false, Some(schema.clone()))?;
813
814        // 'SELECT *' modifiers
815        let mut select_modifiers = SelectModifiers {
816            ilike: None,
817            exclude: PlHashSet::new(),
818            rename: PlHashMap::new(),
819            replace: vec![],
820        };
821
822        let projections = self.column_projections(select_stmt, &schema, &mut select_modifiers)?;
823
824        // Check for "GROUP BY ..." (after determining projections)
825        let mut group_by_keys: Vec<Expr> = Vec::new();
826        match &select_stmt.group_by {
827            // Standard "GROUP BY x, y, z" syntax (also recognising ordinal values)
828            GroupByExpr::Expressions(group_by_exprs, modifiers) => {
829                if !modifiers.is_empty() {
830                    polars_bail!(SQLInterface: "GROUP BY does not support CUBE, ROLLUP, or TOTALS modifiers")
831                }
832                // translate the group expressions, allowing ordinal values
833                group_by_keys = group_by_exprs
834                    .iter()
835                    .map(|e| {
836                        self.expr_or_ordinal(
837                            e,
838                            &projections,
839                            None,
840                            Some(schema.deref()),
841                            "GROUP BY",
842                        )
843                    })
844                    .collect::<PolarsResult<_>>()?
845            },
846            // "GROUP BY ALL" syntax; automatically adds expressions that do not contain
847            // nested agg/window funcs to the group key (also ignores literals).
848            GroupByExpr::All(modifiers) => {
849                if !modifiers.is_empty() {
850                    polars_bail!(SQLInterface: "GROUP BY does not support CUBE, ROLLUP, or TOTALS modifiers")
851                }
852                projections.iter().for_each(|expr| match expr {
853                    // immediately match the most common cases (col|agg|len|lit, optionally aliased).
854                    Expr::Agg(_) | Expr::Len | Expr::Literal(_) => (),
855                    Expr::Column(_) => group_by_keys.push(expr.clone()),
856                    Expr::Alias(e, _)
857                        if matches!(&**e, Expr::Agg(_) | Expr::Len | Expr::Literal(_)) => {},
858                    Expr::Alias(e, _) if matches!(&**e, Expr::Column(_)) => {
859                        if let Expr::Column(name) = &**e {
860                            group_by_keys.push(col(name.clone()));
861                        }
862                    },
863                    _ => {
864                        // If not quick-matched, add if no nested agg/window expressions
865                        if !has_expr(expr, |e| {
866                            matches!(e, Expr::Agg(_))
867                                || matches!(e, Expr::Len)
868                                || matches!(e, Expr::Over { .. })
869                                || {
870                                    #[cfg(feature = "dynamic_group_by")]
871                                    {
872                                        matches!(e, Expr::Rolling { .. })
873                                    }
874                                    #[cfg(not(feature = "dynamic_group_by"))]
875                                    {
876                                        false
877                                    }
878                                }
879                        }) {
880                            group_by_keys.push(expr.clone())
881                        }
882                    },
883                });
884            },
885        };
886
887        lf = if group_by_keys.is_empty() {
888            // The 'having' clause is only valid inside 'group by'
889            if select_stmt.having.is_some() {
890                polars_bail!(SQLSyntax: "HAVING clause not valid outside of GROUP BY; found:\n{:?}", select_stmt.having);
891            };
892
893            // Final/selected cols, accounting for 'SELECT *' modifiers
894            let mut retained_cols = Vec::with_capacity(projections.len());
895            let mut retained_names = Vec::with_capacity(projections.len());
896            let have_order_by = query.order_by.is_some();
897            // Initialize containing InheritsContext to handle empty projection case.
898            let mut projection_heights = ExprSqlProjectionHeightBehavior::InheritsContext;
899
900            // Note: if there is an 'order by' then we project everything (original cols
901            // and new projections) and *then* select the final cols; the retained cols
902            // are used to ensure a correct final projection. If there's no 'order by',
903            // clause then we can project the final column *expressions* directly.
904            for p in projections.iter() {
905                let name = p.to_field(schema.deref())?.name.to_string();
906                if select_modifiers.matches_ilike(&name)
907                    && !select_modifiers.exclude.contains(&name)
908                {
909                    projection_heights |= ExprSqlProjectionHeightBehavior::identify_from_expr(p);
910
911                    retained_cols.push(if have_order_by {
912                        col(name.as_str())
913                    } else {
914                        p.clone()
915                    });
916                    retained_names.push(col(name));
917                }
918            }
919
920            // Apply the remaining modifiers and establish the final projection
921            if have_order_by {
922                // We can safely use `with_columns()` and avoid a join if:
923                // * There is already a projection that projects to the table height.
924                // * All projection heights inherit from context (e.g. all scalar literals that
925                //   are to be broadcasted to table height).
926                if projection_heights.contains(ExprSqlProjectionHeightBehavior::MaintainsColumn)
927                    || projection_heights == ExprSqlProjectionHeightBehavior::InheritsContext
928                {
929                    lf = lf.with_columns(projections);
930                } else {
931                    // We hit this branch if the output height is not guaranteed to match the table
932                    // height. E.g.:
933                    //
934                    // * SELECT COUNT(*) FROM df ORDER BY sort_key;
935                    // * SELECT UNNEST(list_col) FROM df ORDER BY sort_key;
936                    //
937                    // For these cases we truncate / extend the sorting columns with NULLs to match
938                    // the output height. We do this by projecting independently and then joining
939                    // back the original frame on the row index.
940                    const NAME: PlSmallStr = PlSmallStr::from_static("__PL_INDEX");
941                    lf = lf
942                        .clone()
943                        .select(projections)
944                        .with_row_index(NAME, None)
945                        .join(
946                            lf.with_row_index(NAME, None),
947                            [col(NAME)],
948                            [col(NAME)],
949                            JoinArgs {
950                                how: JoinType::Left,
951                                validation: Default::default(),
952                                suffix: None,
953                                slice: None,
954                                nulls_equal: false,
955                                coalesce: Default::default(),
956                                maintain_order: polars_ops::frame::MaintainOrderJoin::Left,
957                            },
958                        );
959                }
960            }
961
962            if !select_modifiers.replace.is_empty() {
963                lf = lf.with_columns(&select_modifiers.replace);
964            }
965            if !select_modifiers.rename.is_empty() {
966                lf = lf.with_columns(select_modifiers.renamed_cols());
967            }
968
969            lf = self.process_order_by(lf, &query.order_by, Some(&retained_cols))?;
970
971            // Note: If `have_order_by`, with_columns is already done above.
972            if projection_heights == ExprSqlProjectionHeightBehavior::InheritsContext
973                && !have_order_by
974            {
975                // All projections need to be broadcasted to table height, so evaluate in `with_columns()`
976                lf = lf.with_columns(retained_cols).select(retained_names);
977            } else {
978                lf = lf.select(retained_cols);
979            }
980
981            if !select_modifiers.rename.is_empty() {
982                lf = lf.rename(
983                    select_modifiers.rename.keys(),
984                    select_modifiers.rename.values(),
985                    true,
986                );
987            };
988            lf
989        } else {
990            lf = self.process_group_by(lf, &group_by_keys, &projections)?;
991            lf = self.process_order_by(lf, &query.order_by, None)?;
992
993            // Apply optional 'having' clause, post-aggregation.
994            let schema = Some(self.get_frame_schema(&mut lf)?);
995            match select_stmt.having.as_ref() {
996                Some(expr) => lf.filter(parse_sql_expr(expr, self, schema.as_deref())?),
997                None => lf,
998            }
999        };
1000
1001        // Apply optional DISTINCT clause.
1002        lf = match &select_stmt.distinct {
1003            Some(Distinct::Distinct) => lf.unique_stable(None, UniqueKeepStrategy::Any),
1004            Some(Distinct::On(exprs)) => {
1005                // TODO: support exprs in `unique` see https://github.com/pola-rs/polars/issues/5760
1006                let schema = Some(self.get_frame_schema(&mut lf)?);
1007                let cols = exprs
1008                    .iter()
1009                    .map(|e| {
1010                        let expr = parse_sql_expr(e, self, schema.as_deref())?;
1011                        if let Expr::Column(name) = expr {
1012                            Ok(name)
1013                        } else {
1014                            Err(polars_err!(SQLSyntax:"DISTINCT ON only supports column names"))
1015                        }
1016                    })
1017                    .collect::<PolarsResult<Vec<_>>>()?;
1018
1019                // DISTINCT ON has to apply the ORDER BY before the operation.
1020                lf = self.process_order_by(lf, &query.order_by, None)?;
1021                return Ok(lf.unique_stable(
1022                    Some(Selector::ByName {
1023                        names: cols.into(),
1024                        strict: true,
1025                    }),
1026                    UniqueKeepStrategy::First,
1027                ));
1028            },
1029            None => lf,
1030        };
1031        Ok(lf)
1032    }

1034    fn column_projections(
1035        &mut self,
1036        select_stmt: &Select,
1037        schema: &SchemaRef,
1038        select_modifiers: &mut SelectModifiers,
1039    ) -> PolarsResult<Vec<Expr>> {
1040        let parsed_items: PolarsResult<Vec<Vec<Expr>>> = select_stmt
1041            .projection
1042            .iter()
1043            .map(|select_item| match select_item {
1044                SelectItem::UnnamedExpr(expr) => {
1045                    Ok(vec![parse_sql_expr(expr, self, Some(schema))?])
1046                },
1047                SelectItem::ExprWithAlias { expr, alias } => {
1048                    let expr = parse_sql_expr(expr, self, Some(schema))?;
1049                    Ok(vec![expr.alias(PlSmallStr::from_str(alias.value.as_str()))])
1050                },
1051                SelectItem::QualifiedWildcard(obj_name, wildcard_options) => self
1052                    .process_qualified_wildcard(
1053                        obj_name,
1054                        wildcard_options,
1055                        select_modifiers,
1056                        Some(schema),
1057                    ),
1058                SelectItem::Wildcard(wildcard_options) => {
1059                    let cols = schema
1060                        .iter_names()
1061                        .map(|name| col(name.clone()))
1062                        .collect::<Vec<_>>();
1063
1064                    self.process_wildcard_additional_options(
1065                        cols,
1066                        wildcard_options,
1067                        select_modifiers,
1068                        Some(schema),
1069                    )
1070                },
1071            })
1072            .collect();
1073
1074        let flattened_exprs: Vec<Expr> = parsed_items?
1075            .into_iter()
1076            .flatten()
1077            .flat_map(|expr| expand_exprs(expr, schema))
1078            .collect();
1079
1080        Ok(flattened_exprs)
1081    }

1083    fn process_where(
1084        &mut self,
1085        mut lf: LazyFrame,
1086        expr: &Option<SQLExpr>,
1087        invert_filter: bool,
1088        schema: Option<SchemaRef>,
1089    ) -> PolarsResult<LazyFrame> {
1090        if let Some(expr) = expr {
1091            let schema = match schema {
1092                None => self.get_frame_schema(&mut lf)?,
1093                Some(s) => s,
1094            };
1095
1096            // shortcut filter evaluation if given expression is just TRUE or FALSE
1097            let (all_true, all_false) = match expr {
1098                SQLExpr::Value(SQLValue::Boolean(b)) => (*b, !*b),
1099                SQLExpr::BinaryOp { left, op, right } => match (&**left, &**right, op) {
1100                    (SQLExpr::Value(a), SQLExpr::Value(b), BinaryOperator::Eq) => (a == b, a != b),
1101                    (SQLExpr::Value(a), SQLExpr::Value(b), BinaryOperator::NotEq) => {
1102                        (a != b, a == b)
1103                    },
1104                    _ => (false, false),
1105                },
1106                _ => (false, false),
1107            };
1108            if (all_true && !invert_filter) || (all_false && invert_filter) {
1109                return Ok(lf);
1110            } else if (all_false && !invert_filter) || (all_true && invert_filter) {
1111                return Ok(DataFrame::empty_with_schema(schema.as_ref()).lazy());
1112            }
1113
1114            // ...otherwise parse and apply the filter as normal
1115            let mut filter_expression = parse_sql_expr(expr, self, Some(schema).as_deref())?;
1116            if filter_expression.clone().meta().has_multiple_outputs() {
1117                filter_expression = all_horizontal([filter_expression])?;
1118            }
1119            lf = self.process_subqueries(lf, vec![&mut filter_expression]);
1120            lf = if invert_filter {
1121                lf.remove(filter_expression)
1122            } else {
1123                lf.filter(filter_expression)
1124            };
1125        }
1126        Ok(lf)
1127    }

1129    pub(super) fn process_join(
1130        &mut self,
1131        tbl_left: &TableInfo,
1132        tbl_right: &TableInfo,
1133        constraint: &JoinConstraint,
1134        join_type: JoinType,
1135    ) -> PolarsResult<LazyFrame> {
1136        let (left_on, right_on) = process_join_constraint(constraint, tbl_left, tbl_right, self)?;
1137        let joined = tbl_left
1138            .frame
1139            .clone()
1140            .join_builder()
1141            .with(tbl_right.frame.clone())
1142            .left_on(left_on)
1143            .right_on(right_on)
1144            .how(join_type)
1145            .suffix(format!(":{}", tbl_right.name))
1146            .coalesce(JoinCoalesce::KeepColumns)
1147            .finish();
1148
1149        Ok(joined)
1150    }

1152    fn process_subqueries(&self, lf: LazyFrame, exprs: Vec<&mut Expr>) -> LazyFrame {
1153        let mut contexts = vec![];
1154        for expr in exprs {
1155            *expr = expr.clone().map_expr(|e| match e {
1156                Expr::SubPlan(lp, names) => {
1157                    contexts.push(<LazyFrame>::from((**lp).clone()));
1158                    if names.len() == 1 {
1159                        Expr::Column(names[0].as_str().into())
1160                    } else {
1161                        Expr::SubPlan(lp, names)
1162                    }
1163                },
1164                e => e,
1165            })
1166        }
1167        if contexts.is_empty() {
1168            lf
1169        } else {
1170            lf.with_context(contexts)
1171        }
1172    }

1174    fn execute_create_table(&mut self, stmt: &Statement) -> PolarsResult<LazyFrame> {
1175        if let Statement::CreateTable(CreateTable {
1176            if_not_exists,
1177            name,
1178            query,
1179            columns,
1180            like,
1181            ..
1182        }) = stmt
1183        {
1184            let tbl_name = name.0.first().unwrap().value.as_str();
1185            if *if_not_exists && self.table_map.contains_key(tbl_name) {
1186                polars_bail!(SQLInterface: "relation '{}' already exists", tbl_name);
1187            }
1188            let lf = match (query, columns.is_empty(), like) {
1189                (Some(query), true, None) => {
1190                    // ----------------------------------------------------
1191                    // CREATE TABLE [IF NOT EXISTS] <name> AS <query>
1192                    // ----------------------------------------------------
1193                    self.execute_query(query)?
1194                },
1195                (None, false, None) => {
1196                    // ----------------------------------------------------
1197                    // CREATE TABLE [IF NOT EXISTS] <name> (<coldef>, ...)
1198                    // ----------------------------------------------------
1199                    let mut schema = Schema::with_capacity(columns.len());
1200                    for col in columns {
1201                        let col_name = col.name.value.as_str();
1202                        let dtype = map_sql_dtype_to_polars(&col.data_type)?;
1203                        schema.insert_at_index(schema.len(), col_name.into(), dtype)?;
1204                    }
1205                    DataFrame::empty_with_schema(&schema).lazy()
1206                },
1207                (None, true, Some(from_name)) => {
1208                    // ----------------------------------------------------
1209                    // CREATE TABLE [IF NOT EXISTS] <name> LIKE <table>
1210                    // ----------------------------------------------------
1211                    let like_table = from_name.0.first().unwrap().value.as_str();
1212                    if let Some(mut table) = self.table_map.get(like_table).cloned() {
1213                        let schema = self.get_frame_schema(&mut table)?;
1214                        DataFrame::empty_with_schema(&schema).lazy()
1215                    } else {
1216                        polars_bail!(SQLInterface: "table given in LIKE does not exist: {}", like_table)
1217                    }
1218                },
1219                // No valid options provided
1220                (None, true, None) => {
1221                    polars_bail!(SQLInterface: "CREATE TABLE expected a query, column definitions, or LIKE clause")
1222                },
1223                // Mutually exclusive options
1224                _ => {
1225                    polars_bail!(
1226                        SQLInterface: "CREATE TABLE received mutually exclusive options:\nquery = {:?}\ncolumns = {:?}\nlike = {:?}",
1227                        query,
1228                        columns,
1229                        like,
1230                    )

1243    fn get_table(&mut self, relation: &TableFactor) -> PolarsResult<(String, LazyFrame)> {
1244        match relation {
1245            TableFactor::Table {
1246                name, alias, args, ..
1247            } => {
1248                if let Some(args) = args {
1249                    return self.execute_table_function(name, alias, &args.args);
1250                }
1251                let tbl_name = name.0.first().unwrap().value.as_str();
1252                if let Some(lf) = self.get_table_from_current_scope(tbl_name) {
1253                    match alias {
1254                        Some(alias) => {
1255                            self.table_aliases
1256                                .insert(alias.name.value.clone(), tbl_name.to_string());
1257                            Ok((alias.to_string(), lf))
1258                        },
1259                        None => Ok((tbl_name.to_string(), lf)),
1260                    }
1261                } else {
1262                    polars_bail!(SQLInterface: "relation '{}' was not found", tbl_name);
1263                }
1264            },
1265            TableFactor::Derived {
1266                lateral,
1267                subquery,
1268                alias,
1269            } => {
1270                polars_ensure!(!(*lateral), SQLInterface: "LATERAL not supported");
1271                if let Some(alias) = alias {
1272                    let mut lf = self.execute_query_no_ctes(subquery)?;
1273                    lf = self.rename_columns_from_table_alias(lf, alias)?;
1274                    self.table_map.insert(alias.name.value.clone(), lf.clone());
1275                    Ok((alias.name.value.clone(), lf))
1276                } else {
1277                    polars_bail!(SQLSyntax: "derived tables must have aliases");
1278                }
1279            },
1280            TableFactor::UNNEST {
1281                alias,
1282                array_exprs,
1283                with_offset,
1284                with_offset_alias: _,
1285                ..
1286            } => {
1287                if let Some(alias) = alias {
1288                    let column_names: Vec<Option<PlSmallStr>> = alias
1289                        .columns
1290                        .iter()
1291                        .map(|c| {
1292                            if c.name.value.is_empty() {
1293                                None
1294                            } else {
1295                                Some(PlSmallStr::from_str(c.name.value.as_str()))
1296                            }
1297                        })
1298                        .collect();
1299
1300                    let column_values: Vec<Series> = array_exprs
1301                        .iter()
1302                        .map(|arr| parse_sql_array(arr, self))
1303                        .collect::<Result<_, _>>()?;
1304
1305                    polars_ensure!(!column_names.is_empty(),
1306                        SQLSyntax:
1307                        "UNNEST table alias must also declare column names, eg: {} (a,b,c)", alias.name.to_string()
1308                    );
1309                    if column_names.len() != column_values.len() {
1310                        let plural = if column_values.len() > 1 { "s" } else { "" };
1311                        polars_bail!(
1312                            SQLSyntax:
1313                            "UNNEST table alias requires {} column name{}, found {}", column_values.len(), plural, column_names.len()
1314                        );
1315                    }
1316                    let column_series: Vec<Column> = column_values
1317                        .into_iter()
1318                        .zip(column_names)
1319                        .map(|(s, name)| {
1320                            if let Some(name) = name {
1321                                s.with_name(name)
1322                            } else {
1323                                s
1324                            }
1325                        })
1326                        .map(Column::from)
1327                        .collect();
1328
1329                    let lf = DataFrame::new(column_series)?.lazy();
1330
1331                    if *with_offset {
1332                        // TODO: support 'WITH ORDINALITY|OFFSET' modifier.
1333                        polars_bail!(SQLInterface: "UNNEST tables do not (yet) support WITH ORDINALITY|OFFSET");
1334                    }
1335                    let table_name = alias.name.value.clone();
1336                    self.table_map.insert(table_name.clone(), lf.clone());
1337                    Ok((table_name, lf))
1338                } else {
1339                    polars_bail!(SQLSyntax: "UNNEST table must have an alias");
1340                }
1341            },
1342            TableFactor::NestedJoin {
1343                table_with_joins,
1344                alias,
1345            } => {
1346                let lf = self.execute_from_statement(table_with_joins)?;
1347                match alias {
1348                    Some(a) => Ok((a.name.value.clone(), lf)),
1349                    None => Ok(("".to_string(), lf)),
1350                }
1351            },
1352            // Support bare table, optionally with an alias, for now
1353            _ => polars_bail!(SQLInterface: "not yet implemented: {}", relation),
1354        }
1355    }

1376    fn process_order_by(
1377        &mut self,
1378        mut lf: LazyFrame,
1379        order_by: &Option<OrderBy>,
1380        selected: Option<&[Expr]>,
1381    ) -> PolarsResult<LazyFrame> {
1382        if order_by.as_ref().is_none_or(|ob| ob.exprs.is_empty()) {
1383            return Ok(lf);
1384        }
1385        let schema = self.get_frame_schema(&mut lf)?;
1386        let columns_iter = schema.iter_names().map(|e| col(e.clone()));
1387
1388        let order_by = order_by.as_ref().unwrap().exprs.clone();
1389        let mut descending = Vec::with_capacity(order_by.len());
1390        let mut nulls_last = Vec::with_capacity(order_by.len());
1391        let mut by: Vec<Expr> = Vec::with_capacity(order_by.len());
1392
1393        if order_by.len() == 1  // support `ORDER BY ALL` (iff there is no column named 'ALL' in the schema)
1394            && matches!(&order_by[0].expr, SQLExpr::Identifier(ident) if ident.value.to_uppercase() == "ALL" && !schema.iter_names().any(|name| name.to_uppercase() == "ALL"))
1395        {
1396            if let Some(selected) = selected {
1397                by.extend(selected.iter().cloned());
1398            } else {
1399                by.extend(columns_iter);
1400            };
1401            let desc_order = !order_by[0].asc.unwrap_or(true);
1402            nulls_last.resize(by.len(), !order_by[0].nulls_first.unwrap_or(desc_order));
1403            descending.resize(by.len(), desc_order);
1404        } else {
1405            let columns = &columns_iter.collect::<Vec<_>>();
1406            for ob in order_by {
1407                // note: if not specified 'NULLS FIRST' is default for DESC, 'NULLS LAST' otherwise
1408                // https://www.postgresql.org/docs/current/queries-order.html
1409                let desc_order = !ob.asc.unwrap_or(true);
1410                nulls_last.push(!ob.nulls_first.unwrap_or(desc_order));
1411                descending.push(desc_order);
1412
1413                // translate order expression, allowing ordinal values
1414                by.push(self.expr_or_ordinal(
1415                    &ob.expr,
1416                    columns,
1417                    selected,
1418                    Some(&schema),
1419                    "ORDER BY",
1420                )?)
1421            }
1422        }
1423        Ok(lf.sort_by_exprs(
1424            &by,
1425            SortMultipleOptions::default()
1426                .with_order_descending_multi(descending)
1427                .with_nulls_last_multi(nulls_last)
1428                .with_maintain_order(true),
1429        ))
1430    }

1432    fn process_group_by(
1433        &mut self,
1434        mut lf: LazyFrame,
1435        group_by_keys: &[Expr],
1436        projections: &[Expr],
1437    ) -> PolarsResult<LazyFrame> {
1438        let schema_before = self.get_frame_schema(&mut lf)?;
1439        let group_by_keys_schema =
1440            expressions_to_schema(group_by_keys, &schema_before, |duplicate_name: &str| {
1441                format!("group_by keys contained duplicate output name '{duplicate_name}'")
1442            })?;
1443
1444        // Remove the group_by keys as polars adds those implicitly.
1445        let mut aggregation_projection = Vec::with_capacity(projections.len());
1446        let mut projection_overrides = PlHashMap::with_capacity(projections.len());
1447        let mut projection_aliases = PlHashSet::new();
1448        let mut group_key_aliases = PlHashSet::new();
1449
1450        for mut e in projections {
1451            // `Len` represents COUNT(*) so we treat as an aggregation here.
1452            let is_non_group_key_expr = has_expr(e, |e| {
1453                match e {
1454                    Expr::Agg(_) | Expr::Len | Expr::Over { .. } => true,
1455                    #[cfg(feature = "dynamic_group_by")]
1456                    Expr::Rolling { .. } => true,
1457                    Expr::Function { function: func, .. }
1458                        if !matches!(func, FunctionExpr::StructExpr(_)) =>
1459                    {
1460                        // If it's a function call containing a column NOT in the group by keys,
1461                        // we treat it as an aggregation.
1462                        has_expr(e, |e| match e {
1463                            Expr::Column(name) => !group_by_keys_schema.contains(name),
1464                            _ => false,
1465                        })
1466                    },
1467                    _ => false,
1468                }
1469            });
1470
1471            // Note: if simple aliased expression we defer aliasing until after the group_by.
1472            if let Expr::Alias(expr, alias) = e {
1473                if e.clone().meta().is_simple_projection(Some(&schema_before)) {
1474                    group_key_aliases.insert(alias.as_ref());
1475                    e = expr
1476                } else if let Expr::Function {
1477                    function: FunctionExpr::StructExpr(StructFunction::FieldByName(name)),
1478                    ..
1479                } = expr.deref()
1480                {
1481                    projection_overrides
1482                        .insert(alias.as_ref(), col(name.clone()).alias(alias.clone()));
1483                } else if !is_non_group_key_expr && !group_by_keys_schema.contains(alias) {
1484                    projection_aliases.insert(alias.as_ref());
1485                }
1486            }
1487            let field = e.to_field(&schema_before)?;
1488            if group_by_keys_schema.get(&field.name).is_none() && is_non_group_key_expr {
1489                let mut e = e.clone();
1490                if let Expr::Agg(AggExpr::Implode(expr)) = &e {
1491                    e = (**expr).clone();
1492                } else if let Expr::Alias(expr, name) = &e {
1493                    if let Expr::Agg(AggExpr::Implode(expr)) = expr.as_ref() {
1494                        e = (**expr).clone().alias(name.clone());
1495                    }
1496                }
1497                aggregation_projection.push(e);
1498            } else if let Expr::Column(_)
1499            | Expr::Function {
1500                function: FunctionExpr::StructExpr(StructFunction::FieldByName(_)),
1501                ..
1502            } = e
1503            {
1504                // Non-aggregated columns must be part of the GROUP BY clause
1505                if !group_by_keys_schema.contains(&field.name) {
1506                    polars_bail!(SQLSyntax: "'{}' should participate in the GROUP BY clause or an aggregate function", &field.name);
1507                }
1508            }
1509        }
1510        let aggregated = lf.group_by(group_by_keys).agg(&aggregation_projection);
1511        let projection_schema =
1512            expressions_to_schema(projections, &schema_before, |duplicate_name: &str| {
1513                format!("group_by aggregations contained duplicate output name '{duplicate_name}'")
1514            })?;
1515
1516        // A final projection to get the proper order and any deferred transforms/aliases.
1517        let final_projection = projection_schema
1518            .iter_names()
1519            .zip(projections)
1520            .map(|(name, projection_expr)| {
1521                if let Some(expr) = projection_overrides.get(name.as_str()) {
1522                    expr.clone()
1523                } else if group_by_keys_schema.get(name).is_some()
1524                    || projection_aliases.contains(name.as_str())
1525                    || group_key_aliases.contains(name.as_str())
1526                {
1527                    projection_expr.clone()
1528                } else {
1529                    col(name.clone())
1530                }
1531            })
1532            .collect::<Vec<_>>();
1533
1534        Ok(aggregated.select(&final_projection))
1535    }

1596    fn process_wildcard_additional_options(
1597        &mut self,
1598        exprs: Vec<Expr>,
1599        options: &WildcardAdditionalOptions,
1600        modifiers: &mut SelectModifiers,
1601        schema: Option<&Schema>,
1602    ) -> PolarsResult<Vec<Expr>> {
1603        if options.opt_except.is_some() && options.opt_exclude.is_some() {
1604            polars_bail!(SQLInterface: "EXCLUDE and EXCEPT wildcard options cannot be used together (prefer EXCLUDE)")
1605        } else if options.opt_exclude.is_some() && options.opt_ilike.is_some() {
1606            polars_bail!(SQLInterface: "EXCLUDE and ILIKE wildcard options cannot be used together")
1607        }
1608
1609        // SELECT * EXCLUDE
1610        if let Some(items) = &options.opt_exclude {
1611            match items {
1612                ExcludeSelectItem::Single(ident) => {
1613                    modifiers.exclude.insert(ident.value.clone());
1614                },
1615                ExcludeSelectItem::Multiple(idents) => {
1616                    modifiers
1617                        .exclude
1618                        .extend(idents.iter().map(|i| i.value.clone()));
1619                },
1620            };
1621        }
1622
1623        // SELECT * EXCEPT
1624        if let Some(items) = &options.opt_except {
1625            modifiers.exclude.insert(items.first_element.value.clone());
1626            modifiers
1627                .exclude
1628                .extend(items.additional_elements.iter().map(|i| i.value.clone()));
1629        }
1630
1631        // SELECT * ILIKE
1632        if let Some(item) = &options.opt_ilike {
1633            let rx = regex::escape(item.pattern.as_str())
1634                .replace('%', ".*")
1635                .replace('_', ".");
1636
1637            modifiers.ilike = Some(
1638                polars_utils::regex_cache::compile_regex(format!("^(?is){rx}$").as_str()).unwrap(),
1639            );
1640        }
1641
1642        // SELECT * RENAME
1643        if let Some(items) = &options.opt_rename {
1644            let renames = match items {
1645                RenameSelectItem::Single(rename) => std::slice::from_ref(rename),
1646                RenameSelectItem::Multiple(renames) => renames.as_slice(),
1647            };
1648            for rn in renames {
1649                let before = PlSmallStr::from_str(rn.ident.value.as_str());
1650                let after = PlSmallStr::from_str(rn.alias.value.as_str());
1651                if before != after {
1652                    modifiers.rename.insert(before, after);
1653                }
1654            }
1655        }
1656
1657        // SELECT * REPLACE
1658        if let Some(replacements) = &options.opt_replace {
1659            for rp in &replacements.items {
1660                let replacement_expr = parse_sql_expr(&rp.expr, self, schema);
1661                modifiers
1662                    .replace
1663                    .push(replacement_expr?.alias(rp.column_name.value.as_str()));
1664            }
1665        }
1666        Ok(exprs)
1667    }

1732/// Visitor that checks if an expression tree contains a reference to a specific table.
1733struct FindTableIdentifier<'a> {
1734    table_name: &'a str,
1735    found: bool,
1736}
1737
1738impl<'a> Visitor for FindTableIdentifier<'a> {
1739    type Break = ();
1740
1741    fn pre_visit_expr(&mut self, expr: &SQLExpr) -> ControlFlow<Self::Break> {
1742        if let SQLExpr::CompoundIdentifier(idents) = expr {
1743            if idents.len() >= 2 && idents[0].value.as_str() == self.table_name {
1744                self.found = true; // return immediately on first match
1745                return ControlFlow::Break(());
1746            }
1747        }
1748        ControlFlow::Continue(())
1749    }
1750}

1752/// Check if all columns referred to in a Polars expression exist in the given Schema.
1753fn expr_cols_all_in_schema(expr: &Expr, schema: &Schema) -> bool {
1754    let mut found_cols = false;
1755    let mut all_in_schema = true;
1756    for e in expr.into_iter() {
1757        if let Expr::Column(name) = e {
1758            found_cols = true;
1759            if !schema.contains(name.as_str()) {
1760                all_in_schema = false;
1761                break;
1762            }
1763        }
1764    }
1765    found_cols && all_in_schema
1766}

1778/// Determine which parsed join expressions actually belong in `left_om` and which in `right_on`.
1779///
1780/// This needs to be handled carefully because in SQL joins you can write "join on" constraints
1781/// either way round, and in joins with more than two tables you can also join against an earlier
1782/// table (e.g.: you could be joining `df1` to `df2` to `df3`, but the final join condition where
1783/// we join  `df2` to `df3` could refer to `df1.a = df3.b`; this takes a little more work to
1784/// resolve as our native `join` function operates on only two tables at a time.
1785fn determine_left_right_join_on(
1786    ctx: &mut SQLContext,
1787    expr_left: &SQLExpr,
1788    expr_right: &SQLExpr,
1789    tbl_left: &TableInfo,
1790    tbl_right: &TableInfo,
1791    join_schema: &Schema,
1792) -> PolarsResult<(Vec<Expr>, Vec<Expr>)> {
1793    // parse, removing any aliases that may have been added by `resolve_column`
1794    // (called inside `parse_sql_expr`) as we need the actual/underlying col
1795    let left_on = match parse_sql_expr(expr_left, ctx, Some(join_schema))? {
1796        Expr::Alias(inner, _) => Arc::unwrap_or_clone(inner),
1797        e => e,
1798    };
1799    let right_on = match parse_sql_expr(expr_right, ctx, Some(join_schema))? {
1800        Expr::Alias(inner, _) => Arc::unwrap_or_clone(inner),
1801        e => e,
1802    };
1803
1804    // ------------------------------------------------------------------
1805    // simple/typical case: can fully resolve SQL-level table references
1806    // ------------------------------------------------------------------
1807    let left_refs = (
1808        expr_refers_to_table(expr_left, &tbl_left.name),
1809        expr_refers_to_table(expr_left, &tbl_right.name),
1810    );
1811    let right_refs = (
1812        expr_refers_to_table(expr_right, &tbl_left.name),
1813        expr_refers_to_table(expr_right, &tbl_right.name),
1814    );
1815    // if the SQL-level references unambiguously indicate table ownership, we're done
1816    match (left_refs, right_refs) {
1817        // standard: left expr → left table, right expr → right table
1818        ((true, false), (false, true)) => return Ok((vec![left_on], vec![right_on])),
1819        // reversed: left expr → right table, right expr → left table
1820        ((false, true), (true, false)) => return Ok((vec![right_on], vec![left_on])),
1821        // unsupported: one side references *both* tables
1822        ((true, true), _) | (_, (true, true)) if tbl_left.name != tbl_right.name => {
1823            polars_bail!(
1824               SQLInterface: "unsupported join condition: {} side references both '{}' and '{}'",
1825               if left_refs.0 && left_refs.1 {
1826                    "left"
1827                } else {
1828                    "right"
1829                }, tbl_left.name, tbl_right.name
1830            )
1831        },
1832        // fall through to the more involved col/ref resolution
1833        _ => {},
1834    }
1835
1836    // ------------------------------------------------------------------
1837    // more involved: additionally employ schema-based column resolution
1838    // (applies to unqualified columns and/or chained joins)
1839    // ------------------------------------------------------------------
1840    let left_on_cols_in = (
1841        expr_cols_all_in_schema(&left_on, &tbl_left.schema),
1842        expr_cols_all_in_schema(&left_on, &tbl_right.schema),
1843    );
1844    let right_on_cols_in = (
1845        expr_cols_all_in_schema(&right_on, &tbl_left.schema),
1846        expr_cols_all_in_schema(&right_on, &tbl_right.schema),
1847    );
1848    match (left_on_cols_in, right_on_cols_in) {
1849        // each expression's columns exist in exactly one schema
1850        ((true, false), (false, true)) => Ok((vec![left_on], vec![right_on])),
1851        ((false, true), (true, false)) => Ok((vec![right_on], vec![left_on])),
1852        // one expression in both, other only in one; prefer the unique one
1853        ((true, true), (true, false)) => Ok((vec![right_on], vec![left_on])),
1854        ((true, true), (false, true)) => Ok((vec![left_on], vec![right_on])),
1855        ((true, false), (true, true)) => Ok((vec![left_on], vec![right_on])),
1856        ((false, true), (true, true)) => Ok((vec![right_on], vec![left_on])),
1857        // pass through as-is
1858        _ => Ok((vec![left_on], vec![right_on])),
1859    }
1860}
1861
1862fn process_join_on(
1863    ctx: &mut SQLContext,
1864    sql_expr: &SQLExpr,
1865    tbl_left: &TableInfo,
1866    tbl_right: &TableInfo,
1867) -> PolarsResult<(Vec<Expr>, Vec<Expr>)> {
1868    match sql_expr {
1869        SQLExpr::BinaryOp { left, op, right } => match op {
1870            BinaryOperator::And => {
1871                let (mut left_i, mut right_i) = process_join_on(ctx, left, tbl_left, tbl_right)?;
1872                let (mut left_j, mut right_j) = process_join_on(ctx, right, tbl_left, tbl_right)?;
1873                left_i.append(&mut left_j);
1874                right_i.append(&mut right_j);
1875                Ok((left_i, right_i))
1876            },
1877            BinaryOperator::Eq => {
1878                // establish unified schema with cols from both tables; needed for multi/chained
1879                // joins where suffixed intermediary/joined cols aren't in an existing schema.
1880                let mut join_schema =
1881                    Schema::with_capacity(tbl_left.schema.len() + tbl_right.schema.len());
1882                for (name, dtype) in tbl_left.schema.iter() {
1883                    join_schema.insert_at_index(join_schema.len(), name.clone(), dtype.clone())?;
1884                }
1885                for (name, dtype) in tbl_right.schema.iter() {
1886                    if !join_schema.contains(name) {
1887                        join_schema.insert_at_index(
1888                            join_schema.len(),
1889                            name.clone(),
1890                            dtype.clone(),
1891                        )?;
1892                    }
1893                }
1894                determine_left_right_join_on(ctx, left, right, tbl_left, tbl_right, &join_schema)
1895            },
1896            _ => polars_bail!(
1897                // TODO: should be able to support more operators later (via `join_where`?)
1898                SQLInterface: "only equi-join constraints (combined with 'AND') are currently supported; found op = '{:?}'", op
1899            ),
1900        },
1901        SQLExpr::Nested(expr) => process_join_on(ctx, expr, tbl_left, tbl_right),
1902        _ => polars_bail!(
1903            SQLInterface: "only equi-join constraints are currently supported; found expression = {:?}", sql_expr
1904        ),
1905    }
1906}
1907
1908fn process_join_constraint(
1909    constraint: &JoinConstraint,
1910    tbl_left: &TableInfo,
1911    tbl_right: &TableInfo,
1912    ctx: &mut SQLContext,
1913) -> PolarsResult<(Vec<Expr>, Vec<Expr>)> {
1914    match constraint {
1915        JoinConstraint::On(expr @ SQLExpr::BinaryOp { .. }) => {
1916            process_join_on(ctx, expr, tbl_left, tbl_right)
1917        },
1918        JoinConstraint::Using(idents) if !idents.is_empty() => {
1919            let using: Vec<Expr> = idents.iter().map(|id| col(id.value.as_str())).collect();
1920            Ok((using.clone(), using))
1921        },
1922        JoinConstraint::Natural => {
1923            let left_names = tbl_left.schema.iter_names().collect::<PlHashSet<_>>();
1924            let right_names = tbl_right.schema.iter_names().collect::<PlHashSet<_>>();
1925            let on: Vec<Expr> = left_names
1926                .intersection(&right_names)
1927                .map(|&name| col(name.clone()))
1928                .collect();
1929            if on.is_empty() {
1930                polars_bail!(SQLInterface: "no common columns found for NATURAL JOIN")
1931            }
1932            Ok((on.clone(), on))
1933        },
1934        _ => polars_bail!(SQLInterface: "unsupported SQL join constraint:\n{:?}", constraint),
1935    }
1936}

1938bitflags::bitflags! {
1939    /// Bitfield indicating whether there exists a projection with the specified height behavior.
1940    ///
1941    /// Used to help determine whether to execute projections in `select()` or `with_columns()`
1942    /// context.
1943    #[derive(PartialEq)]
1944    struct ExprSqlProjectionHeightBehavior: u8 {
1945        /// Maintains the height of input column(s)
1946        const MaintainsColumn = 1 << 0;
1947        /// Height is independent of input, e.g.:
1948        /// * expressions that change length: e.g. slice, explode, filter, gather etc.
1949        /// * aggregations: count(*), first(), sum() etc.
1950        const Independent = 1 << 1;
1951        /// "Inherits" the height of the context, e.g.:
1952        /// * Scalar literals
1953        const InheritsContext = 1 << 2;
1954    }
1955}
1956
1957impl ExprSqlProjectionHeightBehavior {
1958    fn identify_from_expr(expr: &Expr) -> Self {
1959        let mut has_column = false;
1960        let mut has_independent = false;
1961
1962        for e in expr.into_iter() {
1963            use Expr::*;
1964            has_column |= matches!(e, Column(_) | Selector(_));
1965            has_independent |= match e {
1966                // @TODO: This is broken now with functions.
1967                AnonymousFunction { options, .. } => {
1968                    options.returns_scalar() || !options.is_length_preserving()
1969                },
1970                Literal(v) => !v.is_scalar(),
1971                Explode { .. } | Filter { .. } | Gather { .. } | Slice { .. } => true,
1972                Agg { .. } | Len => true,
1973                _ => false,
1974            }
1975        }
1976        if has_independent {
1977            Self::Independent
1978        } else if has_column {
1979            Self::MaintainsColumn
1980        } else {
1981            Self::InheritsContext
1982        }
1983    }
1984}
