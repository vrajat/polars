1//! This file is complicated because we have complicated escape handling. We want to avoid having
2//! to write down each combination of type & escaping, but we also want the compiler to optimize them
3//! to efficient machine code - so no dynamic dispatch. That means a lot of generics and macros.
4//!
5//! We need to differentiate between several kinds of types, and several kinds of escaping we support:
6//!
7//!  - The simplest escaping mechanism are [`QuoteStyle::Always`] and [`QuoteStyle::Never`].
8//!    For `Never` we just never quote. For `Always` we pass any serializer that never quotes
9//!    to [`quote_serializer()`] then it becomes quoted properly.
10//!  - [`QuoteStyle::Necessary`] (the default) is only relevant for strings and floats with decimal_comma,
11//!    as these are the only types that can have newlines (row separators), commas (default column separators)
12//!    or quotes. String escaping is complicated anyway, and it is all inside [`string_serializer()`].
13//!  - The real complication is [`QuoteStyle::NonNumeric`], that doesn't quote numbers (unless necessary)
14//!    and nulls, and quotes any other thing. The problem is that nulls can be within any type, so we
15//!    need to handle two possibilities of quoting everywhere.
16//!
17//! So in case the chosen style is anything but `NonNumeric`, we statically know for each column except strings
18//! whether it should be quoted (and for strings too when not `Necessary`). There we use
19//! `quote_serializer()` or nothing.
20//!
21//! But to help with `NonNumeric`, each serializer carry the potential to distinguish between nulls and non-nulls,
22//! and quote the latter and not the former. But in order to not have the branch when we statically know the answer,
23//! we have an option to statically disable it with a const generic flag `QUOTE_NON_NULL`. Numbers (that should never
24//! be quoted with `NonNumeric`) just always disable this flag.
25//!
26//! So we have three possibilities:
27//!
28//!  1. A serializer that never quotes. This is a bare serializer with `QUOTE_NON_NULL = false`.
29//!  2. A serializer that always quotes. This is a serializer wrapped with `quote_serializer()`,
30//!     but also with `QUOTE_NON_NULL = false`.
31//!  3. A serializer that quotes only non-nulls. This is a bare serializer with `QUOTE_NON_NULL = true`.
32

64fn make_serializer<'a, T, I: Iterator<Item = Option<T>>, const QUOTE_NON_NULL: bool>(
65    f: impl FnMut(T, &mut Vec<u8>, &SerializeOptions),
66    iter: I,
67    update_array: impl FnMut(&'a dyn Array) -> I,
68) -> impl Serializer<'a> {
69    struct SerializerImpl<F, I, Update, const QUOTE_NON_NULL: bool> {
70        f: F,
71        iter: I,
72        update_array: Update,
73    }
74
75    impl<'a, T, F, I, Update, const QUOTE_NON_NULL: bool> Serializer<'a>
76        for SerializerImpl<F, I, Update, QUOTE_NON_NULL>
77    where
78        F: FnMut(T, &mut Vec<u8>, &SerializeOptions),
79        I: Iterator<Item = Option<T>>,
80        Update: FnMut(&'a dyn Array) -> I,
81    {
82        fn serialize(&mut self, buf: &mut Vec<u8>, options: &SerializeOptions) {
83            let item = self.iter.next().expect(TOO_MANY_MSG);
84            match item {
85                Some(item) => {
86                    if QUOTE_NON_NULL {
87                        buf.push(options.quote_char);
88                    }
89                    (self.f)(item, buf, options);
90                    if QUOTE_NON_NULL {
91                        buf.push(options.quote_char);
92                    }
93                },
94                None => buf.extend_from_slice(options.null.as_bytes()),
95            }
96        }
97
98        fn update_array(&mut self, array: &'a dyn Array) {
99            self.iter = (self.update_array)(array);
100        }
101    }
102
103    SerializerImpl::<_, _, _, QUOTE_NON_NULL> {
104        f,
105        iter,
106        update_array,
107    }
108}

416type ChronoFormatIter<'a, 'b> = std::slice::Iter<'a, chrono::format::Item<'b>>;

418#[cfg(any(feature = "dtype-date", feature = "dtype-time"))]
419fn date_and_time_serializer<'a, Underlying: NativeType, T: std::fmt::Display>(
420    format_str: &'a Option<String>,
421    description: &str,
422    array: &'a dyn Array,
423    sample_value: T,
424    mut convert: impl FnMut(Underlying) -> T + Send + 'a,
425    mut format_fn: impl for<'b> FnMut(
426        &T,
427        ChronoFormatIter<'b, 'a>,
428    ) -> chrono::format::DelayedFormat<ChronoFormatIter<'b, 'a>>
429    + Send
430    + 'a,
431    options: &SerializeOptions,
432) -> PolarsResult<Box<dyn Serializer<'a> + Send + 'a>> {
433    let array = array.as_any().downcast_ref().unwrap();
434    let serializer = match format_str {
435        Some(format_str) => {
436            let format = chrono::format::StrftimeItems::new(format_str).parse().map_err(
437                |_| polars_err!(ComputeError: "cannot format {description} with format '{format_str}'"),
438            )?;
439            use std::fmt::Write;
440            // Fail fast for invalid format. This return error faster to the user, and allows us to not return
441            // `Result` from `serialize()`.
442            write!(IgnoreFmt, "{}", format_fn(&sample_value, format.iter())).map_err(
443                |_| polars_err!(ComputeError: "cannot format {description} with format '{format_str}'"),
444            )?;
445            let callback = move |item, buf: &mut Vec<u8>| {
446                let item = convert(item);
447                // We checked the format is valid above.
448                let _ = write!(buf, "{}", format_fn(&item, format.iter()));
449            };
450            date_and_time_final_serializer(array, callback, options)
451        },
452        None => {
453            let callback = move |item, buf: &mut Vec<u8>| {
454                let item = convert(item);
455                // Formatting dates into `Vec<u8>` cannot fail.
456                let _ = write!(buf, "{item}");
457            };
458            date_and_time_final_serializer(array, callback, options)
459        },
460    };
461    Ok(serializer)
462}

464#[cfg(any(
465    feature = "dtype-date",

469fn date_and_time_final_serializer<'a, T: NativeType>(
470    array: &'a PrimitiveArray<T>,
471    callback: impl FnMut(T, &mut Vec<u8>) + Send + 'a,
472    options: &SerializeOptions,
473) -> Box<dyn Serializer<'a> + Send + 'a> {
474    match options.quote_style {
475        QuoteStyle::Always => Box::new(quote_serializer(callback_serializer::<T, false>(
476            array, callback,
477        ))) as Box<dyn Serializer + Send>,
478        QuoteStyle::NonNumeric => Box::new(callback_serializer::<T, true>(array, callback)),
479        _ => Box::new(callback_serializer::<T, false>(array, callback)),
480    }
481}

483pub(super) fn string_serializer<'a, Iter: Send + 'a>(
484    mut f: impl FnMut(&mut Iter) -> Option<&str> + Send + 'a,
485    options: &SerializeOptions,
486    mut update: impl FnMut(&'a dyn Array) -> Iter + Send + 'a,
487    array: &'a dyn Array,
488) -> Box<dyn Serializer<'a> + 'a + Send> {
489    const LF: u8 = b'\n';
490    const CR: u8 = b'\r';
491
492    struct StringSerializer<F, Iter, Update> {
493        serialize: F,
494        update: Update,
495        iter: Iter,
496    }
497
498    impl<'a, F, Iter, Update> Serializer<'a> for StringSerializer<F, Iter, Update>
499    where
500        F: FnMut(&mut Iter, &mut Vec<u8>, &SerializeOptions),
501        Update: FnMut(&'a dyn Array) -> Iter,
502    {
503        fn serialize(&mut self, buf: &mut Vec<u8>, options: &SerializeOptions) {
504            (self.serialize)(&mut self.iter, buf, options);
505        }
506
507        fn update_array(&mut self, array: &'a dyn Array) {
508            self.iter = (self.update)(array);
509        }
510    }
511
512    fn serialize_str_escaped(buf: &mut Vec<u8>, s: &[u8], quote_char: u8, quoted: bool) {
513        let mut iter = memchr_iter(quote_char, s);
514        let first_quote = iter.next();
515        match first_quote {
516            None => buf.extend_from_slice(s),
517            Some(mut quote_pos) => {
518                if !quoted {
519                    buf.push(quote_char);
520                }
521                let mut start_pos = 0;
522                loop {
523                    buf.extend_from_slice(&s[start_pos..quote_pos]);
524                    buf.extend_from_slice(&[quote_char, quote_char]);
525                    match iter.next() {
526                        Some(quote) => {
527                            start_pos = quote_pos + 1;
528                            quote_pos = quote;
529                        },
530                        None => {
531                            buf.extend_from_slice(&s[quote_pos + 1..]);
532                            break;
533                        },
534                    }
535                }
536                if !quoted {
537                    buf.push(quote_char);
538                }
539            },
540        }
541    }
542
543    let iter = update(array);
544    match options.quote_style {
545        QuoteStyle::Always => {
546            let serialize =
547                move |iter: &mut Iter, buf: &mut Vec<u8>, options: &SerializeOptions| {
548                    let quote_char = options.quote_char;
549                    buf.push(quote_char);
550                    let Some(s) = f(iter) else {
551                        buf.extend_from_slice(options.null.as_bytes());
552                        buf.push(quote_char);
553                        return;
554                    };
555                    serialize_str_escaped(buf, s.as_bytes(), quote_char, true);
556                    buf.push(quote_char);
557                };
558            Box::new(StringSerializer {
559                serialize,
560                update,
561                iter,
562            })
563        },
564        QuoteStyle::NonNumeric => {
565            let serialize =
566                move |iter: &mut Iter, buf: &mut Vec<u8>, options: &SerializeOptions| {
567                    let Some(s) = f(iter) else {
568                        buf.extend_from_slice(options.null.as_bytes());
569                        return;
570                    };
571                    let quote_char = options.quote_char;
572                    buf.push(quote_char);
573                    serialize_str_escaped(buf, s.as_bytes(), quote_char, true);
574                    buf.push(quote_char);
575                };
576            Box::new(StringSerializer {
577                serialize,
578                update,
579                iter,
580            })
581        },
582        QuoteStyle::Necessary => {
583            let serialize =
584                move |iter: &mut Iter, buf: &mut Vec<u8>, options: &SerializeOptions| {
585                    let Some(s) = f(iter) else {
586                        buf.extend_from_slice(options.null.as_bytes());
587                        return;
588                    };
589                    let quote_char = options.quote_char;
590                    // An empty string conflicts with null, so it is necessary to quote.
591                    if s.is_empty() {
592                        buf.extend_from_slice(&[quote_char, quote_char]);
593                        return;
594                    }
595                    let needs_quote = memchr3(options.separator, LF, CR, s.as_bytes()).is_some();
596                    if needs_quote {
597                        buf.push(quote_char);
598                    }
599                    serialize_str_escaped(buf, s.as_bytes(), quote_char, needs_quote);
600                    if needs_quote {
601                        buf.push(quote_char);
602                    }
603                };
604            Box::new(StringSerializer {
605                serialize,
606                update,
607                iter,
608            })
609        },
610        QuoteStyle::Never => {
611            let serialize =
612                move |iter: &mut Iter, buf: &mut Vec<u8>, options: &SerializeOptions| {
613                    let Some(s) = f(iter) else {
614                        buf.extend_from_slice(options.null.as_bytes());
615                        return;
616                    };
617                    buf.extend_from_slice(s.as_bytes());
618                };
619            Box::new(StringSerializer {
620                serialize,
621                update,
622                iter,
623            })
624        },
625    }
626}

628fn quote_serializer<'a>(serializer: impl Serializer<'a>) -> impl Serializer<'a> {
629    struct QuoteSerializer<S>(S);
630    impl<'a, S: Serializer<'a>> Serializer<'a> for QuoteSerializer<S> {
631        fn serialize(&mut self, buf: &mut Vec<u8>, options: &SerializeOptions) {
632            buf.push(options.quote_char);
633            self.0.serialize(buf, options);
634            buf.push(options.quote_char);
635        }
636
637        fn update_array(&mut self, array: &'a dyn Array) {
638            self.0.update_array(array);
639        }
640    }
641    QuoteSerializer(serializer)
642}
643
644pub(super) fn serializer_for<'a>(
645    array: &'a dyn Array,
646    options: &'a SerializeOptions,
647    dtype: &'a DataType,
648    _datetime_format: &'a str,
649    _time_zone: Option<Tz>,
650) -> PolarsResult<Box<dyn Serializer<'a> + Send + 'a>> {
651    // The needs_quotes flag captures the quote logic for the quote_wrapper! macro
652    // It is targeted at numerical types primarily; other types may required additional logic
653    let needs_quotes = match dtype {
654        DataType::Float32 | DataType::Float64 => {
655            // When comma is used as both the field separator and decimal separator, quoting
656            // may be required. Specifically, when:
657            // - quote_style is Always, or
658            // - quote_style is Necessary or Non-Numeric, the field separator is also a comma,
659            //   and the float string field contains a comma character (no precision or precision > 0)
660            //
661            // In some rare cases, a field may get quoted when it is not strictly necessary
662            // (e.g., in scientific notation when only the first digit is non-zero such as '1e12',
663            // or null values in 'non_numeric' quote_style).
664
665            let mut should_quote = options.decimal_comma && options.separator == b',';
666            if let Some(precision) = options.float_precision {
667                should_quote &= precision > 0;
668            }
669
670            match options.quote_style {
671                QuoteStyle::Always => true,
672                QuoteStyle::Necessary | QuoteStyle::NonNumeric => should_quote,
673                QuoteStyle::Never => false,
674            }
675        },
676        #[cfg(feature = "dtype-decimal")]
677        DataType::Decimal(_, scale) => {
678            // Similar to logic for float data-types, but need to consider scale rather than precision
679            let should_quote = options.decimal_comma && options.separator == b',' && *scale > 0;
680
681            match options.quote_style {
682                QuoteStyle::Always => true,
683                QuoteStyle::Necessary | QuoteStyle::NonNumeric => should_quote,
684                QuoteStyle::Never => false,
685            }
686        },
687        _ => options.quote_style == QuoteStyle::Always,
688    };
689
690    macro_rules! quote_wrapper {
691        ($make_serializer:path, $($arg:tt)*) => {{
692            let serializer = $make_serializer(array.as_any().downcast_ref().unwrap(), $($arg)*);
693            if needs_quotes {
694                Box::new(quote_serializer(serializer)) as Box<dyn Serializer + Send>
695            } else {
696                Box::new(serializer)
697            }
698        }};
699        ($make_serializer:path) => { quote_wrapper!($make_serializer,) };
700    }
701
702    let serializer = match dtype {
703        DataType::Int8 => quote_wrapper!(integer_serializer::<i8>),
704        DataType::UInt8 => quote_wrapper!(integer_serializer::<u8>),
705        DataType::Int16 => quote_wrapper!(integer_serializer::<i16>),
706        DataType::UInt16 => quote_wrapper!(integer_serializer::<u16>),
707        DataType::Int32 => quote_wrapper!(integer_serializer::<i32>),
708        DataType::UInt32 => quote_wrapper!(integer_serializer::<u32>),
709        DataType::Int64 => quote_wrapper!(integer_serializer::<i64>),
710        DataType::UInt64 => quote_wrapper!(integer_serializer::<u64>),
711        DataType::Int128 => quote_wrapper!(integer_serializer::<i128>),
712        DataType::UInt128 => quote_wrapper!(integer_serializer::<u128>),
713        DataType::Float32 => {
714            match (
715                options.decimal_comma,
716                options.float_precision,
717                options.float_scientific,
718            ) {
719                // standard decimal separator (period)
720                (false, Some(precision), Some(true)) => {
721                    quote_wrapper!(float_serializer_with_precision_scientific::<f32>, precision)
722                },
723                (false, Some(precision), _) => {
724                    quote_wrapper!(float_serializer_with_precision_positional::<f32>, precision)
725                },
726                (false, None, Some(true)) => {
727                    quote_wrapper!(float_serializer_no_precision_scientific::<f32>)
728                },
729                (false, None, Some(false)) => {
730                    quote_wrapper!(float_serializer_no_precision_positional::<f32>)
731                },
732                (false, None, None) => {
733                    quote_wrapper!(float_serializer_no_precision_autoformat::<f32>)
734                },
735
736                // comma as the decimal separator
737                (true, Some(precision), Some(true)) => quote_wrapper!(
738                    float_serializer_with_precision_scientific_decimal_comma::<f32>,
739                    precision
740                ),
741                (true, Some(precision), _) => quote_wrapper!(
742                    float_serializer_with_precision_positional_decimal_comma::<f32>,
743                    precision
744                ),
745                (true, None, Some(true)) => {
746                    quote_wrapper!(float_serializer_no_precision_scientific_decimal_comma::<f32>)
747                },
748                (true, None, Some(false)) => {
749                    quote_wrapper!(float_serializer_no_precision_positional_decimal_comma::<f32>)
750                },
751                (true, None, None) => {
752                    quote_wrapper!(float_serializer_no_precision_autoformat_decimal_comma::<f32>)
753                },
754            }
755        },
756        DataType::Float64 => {
757            match (
758                options.decimal_comma,
759                options.float_precision,
760                options.float_scientific,
761            ) {
762                // standard decimal separator (period)
763                (false, Some(precision), Some(true)) => {
764                    quote_wrapper!(float_serializer_with_precision_scientific::<f64>, precision)
765                },
766                (false, Some(precision), _) => {
767                    quote_wrapper!(float_serializer_with_precision_positional::<f64>, precision)
768                },
769                (false, None, Some(true)) => {
770                    quote_wrapper!(float_serializer_no_precision_scientific::<f64>)
771                },
772                (false, None, Some(false)) => {
773                    quote_wrapper!(float_serializer_no_precision_positional::<f64>)
774                },
775                (false, None, None) => {
776                    quote_wrapper!(float_serializer_no_precision_autoformat::<f64>)
777                },
778
779                // comma as the decimal separator
780                (true, Some(precision), Some(true)) => quote_wrapper!(
781                    float_serializer_with_precision_scientific_decimal_comma::<f64>,
782                    precision
783                ),
784                (true, Some(precision), _) => quote_wrapper!(
785                    float_serializer_with_precision_positional_decimal_comma::<f64>,
786                    precision
787                ),
788                (true, None, Some(true)) => {
789                    quote_wrapper!(float_serializer_no_precision_scientific_decimal_comma::<f64>)
790                },
791                (true, None, Some(false)) => {
792                    quote_wrapper!(float_serializer_no_precision_positional_decimal_comma::<f64>)
793                },
794                (true, None, None) => {
795                    quote_wrapper!(float_serializer_no_precision_autoformat_decimal_comma::<f64>)
796                },
797            }
798        },
799        DataType::Null => quote_wrapper!(null_serializer),
800        DataType::Boolean => {
801            let array = array.as_any().downcast_ref().unwrap();
802            match options.quote_style {
803                QuoteStyle::Always => Box::new(quote_serializer(bool_serializer::<false>(array)))
804                    as Box<dyn Serializer + Send>,
805                QuoteStyle::NonNumeric => Box::new(bool_serializer::<true>(array)),
806                _ => Box::new(bool_serializer::<false>(array)),
807            }
808        },
809        #[cfg(feature = "dtype-date")]
810        DataType::Date => date_and_time_serializer(
811            &options.date_format,
812            "NaiveDate",
813            array,
814            chrono::NaiveDate::MAX,
815            arrow::temporal_conversions::date32_to_date,
816            |date, items| date.format_with_items(items),
817            options,
818        )?,
819        #[cfg(feature = "dtype-time")]
820        DataType::Time => date_and_time_serializer(
821            &options.time_format,
822            "NaiveTime",
823            array,
824            chrono::NaiveTime::MIN,
825            arrow::temporal_conversions::time64ns_to_time,
826            |time, items| time.format_with_items(items),
827            options,
828        )?,
829        #[cfg(feature = "dtype-datetime")]
830        DataType::Datetime(time_unit, _) => {
831            let format = chrono::format::StrftimeItems::new(_datetime_format)
832                .parse()
833                .map_err(|_| {
834                    polars_err!(
835                        ComputeError: "cannot format {} with format '{_datetime_format}'",
836                        if _time_zone.is_some() { "DateTime" } else { "NaiveDateTime" },
837                    )
838                })?;
839            use std::fmt::Write;
840            let sample_datetime = match _time_zone {
841                #[cfg(feature = "timezones")]
842                Some(time_zone) => time_zone
843                    .from_utc_datetime(&chrono::NaiveDateTime::MAX)
844                    .format_with_items(format.iter()),
845                #[cfg(not(feature = "timezones"))]
846                Some(_) => panic!("activate 'timezones' feature"),
847                None => chrono::NaiveDateTime::MAX.format_with_items(format.iter()),
848            };
849            // Fail fast for invalid format. This return error faster to the user, and allows us to not return
850            // `Result` from `serialize()`.
851            write!(IgnoreFmt, "{sample_datetime}").map_err(|_| {
852                polars_err!(
853                    ComputeError: "cannot format {} with format '{_datetime_format}'",
854                    if _time_zone.is_some() { "DateTime" } else { "NaiveDateTime" },
855                )
856            })?;
857
858            let array = array.as_any().downcast_ref().unwrap();
859
860            macro_rules! time_unit_serializer {
861                ($convert:ident) => {
862                    match _time_zone {
863                        #[cfg(feature = "timezones")]
864                        Some(time_zone) => {
865                            let callback = move |item, buf: &mut Vec<u8>| {
866                                let item = arrow::temporal_conversions::$convert(item);
867                                let item = time_zone.from_utc_datetime(&item);
868                                // We checked the format is valid above.
869                                let _ = write!(buf, "{}", item.format_with_items(format.iter()));
870                            };
871                            date_and_time_final_serializer(array, callback, options)
872                        },
873                        #[cfg(not(feature = "timezones"))]
874                        Some(_) => panic!("activate 'timezones' feature"),
875                        None => {
876                            let callback = move |item, buf: &mut Vec<u8>| {
877                                let item = arrow::temporal_conversions::$convert(item);
878                                // We checked the format is valid above.
879                                let _ = write!(buf, "{}", item.format_with_items(format.iter()));
880                            };
881                            date_and_time_final_serializer(array, callback, options)
882                        },
883                    }
884                };
885            }
886
887            match time_unit {
888                TimeUnit::Nanoseconds => time_unit_serializer!(timestamp_ns_to_datetime),
889                TimeUnit::Microseconds => time_unit_serializer!(timestamp_us_to_datetime),
890                TimeUnit::Milliseconds => time_unit_serializer!(timestamp_ms_to_datetime),
891            }
892        },
893        DataType::String => string_serializer(
894            |iter| Iterator::next(iter).expect(TOO_MANY_MSG),
895            options,
896            |arr| {
897                arr.as_any()
898                    .downcast_ref::<Utf8ViewArray>()
899                    .expect(ARRAY_MISMATCH_MSG)
900                    .iter()
901            },
902            array,
903        ),
904        #[cfg(feature = "dtype-categorical")]
905        DataType::Categorical(_, mapping) | DataType::Enum(_, mapping) => {
906            polars_core::with_match_categorical_physical_type!(dtype.cat_physical().unwrap(), |$C| {
907                string_serializer(
908                    |iter| {
909                        let &idx: &<$C as PolarsCategoricalType>::Native = Iterator::next(iter).expect(TOO_MANY_MSG)?;
910                        Some(unsafe { mapping.cat_to_str_unchecked(idx.as_cat()) })
911                    },
912                    options,
913                    |arr| {
914                        arr.as_any()
915                            .downcast_ref::<PrimitiveArray<<$C as PolarsCategoricalType>::Native>>()
916                            .expect(ARRAY_MISMATCH_MSG)
917                            .iter()
918                    },
919                    array,
920                )
921            })
922        },
923        #[cfg(feature = "dtype-decimal")]
924        DataType::Decimal(_, scale) => {
925            quote_wrapper!(decimal_serializer, *scale)
926        },
927        _ => {
928            polars_bail!(ComputeError: "datatype {dtype} cannot be written to CSV\n\nConsider using JSON or a binary format.")
929        },
930    };
931    Ok(serializer)
932}
